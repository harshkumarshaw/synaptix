## Phase B — Electives UI (90-120 min)

### B.1 Student Preference Ranking

`src/app/(authenticated)/electives/page.tsx` — role-based:
- Student → preference ranking view
- Admin → allocation dashboard

`src/app/(authenticated)/electives/student-preferences.tsx`:

The centrepiece: a **drag-and-drop ranking interface** for each block.

```typescript
"use client";

import { useState } from "react";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { GripVertical, ArrowUp, ArrowDown } from "lucide-react";
import { useElectives, useMyPreferences, useSubmitPreferences } from "@/hooks/use-electives";
import { useToast } from "@/hooks/use-toast";

export function StudentPreferences() {
  const [activeBlock, setActiveBlock] = useState<"Block 1" | "Block 2">("Block 1");
  const { data: electives } = useElectives(activeBlock);
  const { data: existing } = useMyPreferences(activeBlock);
  const submitPrefs = useSubmitPreferences();
  const { toast } = useToast();

  // Ranked list state — initialise from existing preferences or empty
  const [ranked, setRanked] = useState<string[]>([]);

  function moveUp(index: number) {
    if (index === 0) return;
    const next = [...ranked];
    [next[index - 1], next[index]] = [next[index], next[index - 1]];
    setRanked(next);
  }

  function moveDown(index: number) {
    if (index === ranked.length - 1) return;
    const next = [...ranked];
    [next[index], next[index + 1]] = [next[index + 1], next[index]];
    setRanked(next);
  }

  function addToRanking(electiveId: string) {
    if (ranked.includes(electiveId)) return;
    setRanked([...ranked, electiveId]);
  }

  function removeFromRanking(electiveId: string) {
    setRanked(ranked.filter((id) => id !== electiveId));
  }

  async function handleSubmit() {
    const preferences = ranked.map((electiveId, i) => ({
      elective_id: electiveId,
      rank_position: i + 1,
    }));

    submitPrefs.mutate(
      { block: activeBlock, preferences },
      {
        onSuccess: () => toast({ title: "Preferences saved!" }),
        onError: (err: any) => toast({
          title: "Error",
          description: err.response?.data?.detail?.message || "Failed",
          variant: "destructive",
        }),
      }
    );
  }

  const rankedElectives = ranked.map((id) => electives?.find((e) => e.id === id)).filter(Boolean);
  const unranked = electives?.filter((e) => !ranked.includes(e.id)) ?? [];

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold">Elective Preferences</h1>
          <p className="text-muted-foreground">Rank your preferred electives for {activeBlock}</p>
        </div>
        <div className="flex gap-2">
          <Button
            variant={activeBlock === "Block 1" ? "default" : "outline"}
            onClick={() => setActiveBlock("Block 1")}
          >Block 1</Button>
          <Button
            variant={activeBlock === "Block 2" ? "default" : "outline"}
            onClick={() => setActiveBlock("Block 2")}
          >Block 2</Button>
        </div>
      </div>

      <div className="grid gap-6 lg:grid-cols-2">
        {/* Left: Ranked preferences */}
        <Card>
          <CardHeader>
            <CardTitle className="text-base">Your Ranking (drag to reorder)</CardTitle>
          </CardHeader>
          <CardContent>
            {rankedElectives.length === 0 ? (
              <p className="text-sm text-muted-foreground py-4 text-center">
                Add electives from the right panel to build your ranking.
              </p>
            ) : (
              <div className="space-y-2">
                {rankedElectives.map((elective, i) => (
                  <div
                    key={elective!.id}
                    className="flex items-center gap-3 rounded-lg border p-3 bg-background"
                  >
                    <GripVertical className="h-4 w-4 text-muted-foreground cursor-grab" />
                    <Badge variant="secondary" className="w-6 h-6 flex items-center justify-center p-0 text-xs">
                      {i + 1}
                    </Badge>
                    <div className="flex-1">
                      <p className="text-sm font-medium">{elective!.title}</p>
                      <p className="text-xs text-muted-foreground capitalize">{elective!.elective_type}</p>
                    </div>
                    <div className="flex gap-1">
                      <button onClick={() => moveUp(i)} className="p-1 hover:bg-muted rounded" disabled={i === 0}>
                        <ArrowUp className="h-3.5 w-3.5" />
                      </button>
                      <button onClick={() => moveDown(i)} className="p-1 hover:bg-muted rounded" disabled={i === rankedElectives.length - 1}>
                        <ArrowDown className="h-3.5 w-3.5" />
                      </button>
                      <button onClick={() => removeFromRanking(elective!.id)} className="p-1 hover:bg-muted rounded text-destructive">
                        ✕
                      </button>
                    </div>
                  </div>
                ))}
              </div>
            )}

            <Button
              className="w-full mt-4"
              onClick={handleSubmit}
              disabled={ranked.length === 0 || submitPrefs.isPending}
            >
              {submitPrefs.isPending ? "Saving..." : `Submit Ranking (${ranked.length} electives)`}
            </Button>
          </CardContent>
        </Card>

        {/* Right: Available electives */}
        <Card>
          <CardHeader>
            <CardTitle className="text-base">Available Electives</CardTitle>
          </CardHeader>
          <CardContent className="space-y-2">
            {unranked.map((elective) => (
              <div
                key={elective.id}
                className="flex items-center justify-between rounded-lg border p-3 cursor-pointer hover:bg-muted/50"
                onClick={() => addToRanking(elective.id)}
              >
                <div>
                  <p className="text-sm font-medium">{elective.title}</p>
                  <p className="text-xs text-muted-foreground">
                    {elective.elective_type} · {elective.allocated_count}/{elective.capacity} filled
                  </p>
                </div>
                <Button variant="ghost" size="sm">+ Add</Button>
              </div>
            ))}
          </CardContent>
        </Card>
      </div>
    </div>
  );
}
```

```typescript
// Hook: preview attendance impact of a leave approval
export function useLeaveImpactPreview(leaveRequestId: string) {

      return data;
    },
    enabled: !!leaveRequestId,
  });
}
```

```
ADR-049: IA Aggregation Reads Across Phase 2 and Phase 3 Tables

Decision: ia_aggregation service reads:
- logbook_assessments.ia_marks_awarded (Phase 2, existing)
- viva_scores.marks_obtained (Phase 3, new)
- practical_assessments.marks_obtained (Phase 3, new)
- clinical_evaluations.marks_obtained (Phase 3, new)

Weighted aggregation per MDM config. Result stored in ia_aggregation table.
Service-layer implementation (not trigger) because cross-module.
```
